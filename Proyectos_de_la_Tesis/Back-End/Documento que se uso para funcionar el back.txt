Código de la creación de la tabla amigos
	- URL = https://www.troyfawkes.com/solved-mysql-error-1005-cant-create-table-errno-150/
	- CREATE TABLE IF NOT EXISTS unidadfamiliar.app_friend (

	id_friend	INT NOT NULL AUTO_INCREMENT,
    
	id_user	INT UNSIGNED NOT NULL,
    
	name_user VARCHAR(30),
    
	telefono_user VARCHAR(15),
    
	telefono_friend VARCHAR(15),
    
	nombre_friend	VARCHAR(30),
    
	email_friend	VARCHAR(50),
    
	count_friend	INT,
    
	created_friend	DATETIME,
    
	status_friend	CHAR(1),
    
	INDEX idx_user (id_user ASC),
	
	PRIMARY KEY (id_friend),
    
	CONSTRAINT users_const
    
	FOREIGN KEY (id_user) 
    REFERENCES unidadfamiliar.cms_users (id)
    
	ON DELETE NO ACTION
    
	ON UPDATE NO ACTION

) ENGINE = InnoDB; 

Código de la creacion de la tabla de posición
	- CREATE TABLE IF NOT EXISTS unidadfamiliar.app_position (
	
	id_position	INT NOT NULL AUTO_INCREMENT,
    
	id_user	INT UNSIGNED NOT NULL,
	id_friend	INT NOT NULL,
    
	name_user VARCHAR(30) NOT NULL,
    
	latitud_position DECIMAL NOT NULL,
    
	longitud_position DECIMAL NOT NULL,
    
	direccion_position	VARCHAR(250) NOT NULL,
    
	nombre_friend	VARCHAR(30) NOT NULL,
    
	created_position	TIMESTAMP,
    
	updated_position    TIMESTAMP,
    
	status_position	CHAR(1),
    
	INDEX idx_friend (id_friend ASC),
	
	PRIMARY KEY (id_position),
    
	CONSTRAINT friend_const
    
	FOREIGN KEY (id_friend) 
    REFERENCES unidadfamiliar.app_friend (id_friend)
    
	ON DELETE NO ACTION
    
	ON UPDATE NO ACTION

) ENGINE = InnoDB;
